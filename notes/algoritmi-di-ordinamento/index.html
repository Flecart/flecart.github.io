    private static int partition(Comparable A[], int i, int f) {
	int inf = i, sup = f + 1;
	Comparable temp, x = A[i];
	while (true) {
		do {

		} while (inf <= f && A[inf].compareTo(x) <= 0);

			A[sup] = temp;

	A[sup] = temp;

}
